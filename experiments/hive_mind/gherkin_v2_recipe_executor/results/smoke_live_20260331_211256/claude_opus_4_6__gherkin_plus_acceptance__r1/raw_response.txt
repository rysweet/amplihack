## `recipe_step_executor.py`

```python
"""Recipe Step Executor — executes multi-step workflows with conditions,
dependencies, retries, timeouts, output capture, and sub-recipes.

Public API:
    RecipeStepExecutor: Main executor class
    StepResult: Result of a single step execution
"""

from __future__ import annotations

import asyncio
import re
import time
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Callable

__all__ = ["RecipeStepExecutor", "StepResult", "StepStatus"]


class StepStatus(str, Enum):
    COMPLETED = "completed"
    FAILED = "failed"
    SKIPPED = "skipped"
    TIMED_OUT = "timed_out"


@dataclass
class StepResult:
    step_id: str
    status: StepStatus
    output: str | None = None
    attempt_count: int = 1
    failure_reason: str | None = None
    start_time: float | None = None
    end_time: float | None = None
    retry_delays: list[float] = field(default_factory=list)

    @property
    def execution_time(self) -> float | None:
        if self.start_time is not None and self.end_time is not None:
            return self.end_time - self.start_time
        return None


class RecipeStepExecutor:
    """Executes recipe steps with conditions, dependencies, retries,
    timeouts, output capture, and sub-recipe delegation."""

    def __init__(
        self,
        command_runner: Callable[[str, dict, float], tuple[int, str]] | None = None,
    ):
        """Initialize executor.

        Args:
            command_runner: Optional callable(command, context, timeout) -> (exit_code, stdout).
                           If None, a default shell-like runner is used.
        """
        self._command_runner = command_runner or self._default_command_runner
        self._results: dict[str, StepResult] = {}

    def execute(self, recipe: list[dict], context: dict) -> dict:
        """Execute a recipe (list of step dicts) against a context.

        Returns:
            dict with keys:
                - 'context': the final execution context
                - 'results': dict mapping step_id -> StepResult
        """
        self._results = {}
        self._execute_steps(recipe, context)
        return {"context": context, "results": dict(self._results)}

    def _execute_steps(self, steps: list[dict], context: dict) -> None:
        """Execute steps respecting dependency ordering."""
        step_map = {s["id"]: s for s in steps}
        executed: set[str] = set()

        # Topological execution: keep iterating until all steps are resolved
        remaining = list(steps)
        while remaining:
            progress = False
            next_remaining = []
            for step in remaining:
                sid = step["id"]
                deps = self._parse_deps(step.get("blockedBy", ""))
                # Check if all deps are resolved
                if all(d in executed for d in deps):
                    self._execute_single_step(step, context, step_map)
                    executed.add(sid)
                    progress = True
                else:
                    next_remaining.append(step)
            if not progress:
                # Circular dependency or unresolvable — fail remaining
                for step in next_remaining:
                    sid = step["id"]
                    self._results[sid] = StepResult(
                        step_id=sid,
                        status=StepStatus.FAILED,
                        failure_reason="unresolvable_dependency",
                    )
                    executed.add(sid)
                break
            remaining = next_remaining

    def _execute_single_step(
        self, step: dict, context: dict, step_map: dict
    ) -> None:
        sid = step["id"]
        deps = self._parse_deps(step.get("blockedBy", ""))

        # --- Dependency check BEFORE condition check ---
        for dep_id in deps:
            dep_result = self._results.get(dep_id)
            if dep_result and dep_result.status in (
                StepStatus.FAILED,
                StepStatus.TIMED_OUT,
            ):
                self._results[sid] = StepResult(
                    step_id=sid,
                    status=StepStatus.FAILED,
                    failure_reason="dependency_failed",
                )
                return

        # --- Condition check ---
        condition = step.get("condition", "").strip()
        if condition and not self._evaluate_condition(condition, context):
            self._results[sid] = StepResult(
                step_id=sid, status=StepStatus.SKIPPED
            )
            return

        # --- Sub-recipe handling ---
        sub_recipe = step.get("sub_recipe")
        if sub_recipe is not None:
            self._execute_sub_recipe(step, context)
            return

        # --- Command execution with retry/timeout ---
        command = step.get("command", "")
        max_retries = int(step.get("max_retries", 0))
        timeout = float(step.get("timeout_seconds", 60))

        result = self._run_with_retries(sid, command, context, max_retries, timeout)
        self._results[sid] = result

        if result.status == StepStatus.COMPLETED and result.output is not None:
            context[sid] = result.output

    def _run_with_retries(
        self,
        sid: str,
        command: str,
        context: dict,
        max_retries: int,
        timeout: float,
    ) -> StepResult:
        total_attempts = max_retries + 1
        retry_delays: list[float] = []
        resolved_command = self._resolve_templates(command, context)

        for attempt in range(1, total_attempts + 1):
            if attempt > 1:
                delay = 2 ** (attempt - 2)  # 1, 2, 4, ...
                retry_delays.append(float(delay))
                time.sleep(delay)

            start = time.monotonic()
            try:
                exit_code, stdout = self._command_runner(
                    resolved_command, context, timeout
                )
                end = time.monotonic()
            except TimeoutError:
                end = time.monotonic()
                return StepResult(
                    step_id=sid,
                    status=StepStatus.TIMED_OUT,
                    attempt_count=attempt,
                    start_time=start,
                    end_time=end,
                    retry_delays=retry_delays,
                )

            if exit_code == 0:
                output = stdout.strip() if stdout else ""
                return StepResult(
                    step_id=sid,
                    status=StepStatus.COMPLETED,
                    output=output,
                    attempt_count=attempt,
                    start_time=start,
                    end_time=end,
                    retry_delays=retry_delays,
                )

        return StepResult(
            step_id=sid,
            status=StepStatus.FAILED,
            attempt_count=total_attempts,
            start_time=start,
            end_time=end,
            retry_delays=retry_delays,
        )

    def _execute_sub_recipe(self, step: dict, parent_context: dict) -> None:
        sid = step["id"]
        sub_recipe_data = step["sub_recipe"]
        if isinstance(sub_recipe_data, str):
            import json
            sub_recipe_data = json.loads(sub_recipe_data)

        propagate = str(step.get("propagate_outputs", "false")).lower() == "true"
        max_retries = int(step.get("max_retries", 0))

        # Child context inherits parent
        child_context = dict(parent_context)

        child_executor = RecipeStepExecutor(command_runner=self._command_runner)
        child_result = child_executor.execute(sub_recipe_data, child_context)
        child_results = child_result["results"]

        # Check if any child step failed
        any_failed = any(
            r.status in (StepStatus.FAILED, StepStatus.TIMED_OUT)
            for r in child_results.values()
        )

        if any_failed:
            # Sub-recipe failure is non-transient — never retry
            self._results[sid] = StepResult(
                step_id=sid,
                status=StepStatus.FAILED,
                attempt_count=1,
                failure_reason="sub_recipe_failed",
            )
            return

        # Propagate outputs if requested
        if propagate:
            for child_sid, child_r in child_results.items():
                if child_r.status == StepStatus.COMPLETED and child_r.output is not None:
                    parent_context[child_sid] = child_r.output

        self._results[sid] = StepResult(
            step_id=sid,
            status=StepStatus.COMPLETED,
            attempt_count=1,
        )

    def _evaluate_condition(self, condition: str, context: dict) -> bool:
        """Evaluate a condition string as a Python expression against context."""
        try:
            return bool(eval(condition, {"__builtins__": {}}, dict(context)))
        except Exception:
            return False

    def _resolve_templates(self, text: str, context: dict) -> str:
        """Replace {{key}} with context[key] value. Missing keys remain literal."""
        def replacer(match: re.Match) -> str:
            key = match.group(1).strip()
            if key in context:
                return str(context[key])
            return match.group(0)  # Keep literal
        return re.sub(r"\{\{(\s*\w+\s*)\}\}", replacer, text)

    @staticmethod
    def _parse_deps(blocked_by: str | list) -> list[str]:
        if not blocked_by:
            return []
        if isinstance(blocked_by, list):
            return [d.strip() for d in blocked_by if d.strip()]
        return [d.strip() for d in str(blocked_by).split(",") if d.strip()]

    @staticmethod
    def _default_command_runner(
        command: str, context: dict, timeout: float
    ) -> tuple[int, str]:
        """Default runner using subprocess."""
        import subprocess

        try:
            result = subprocess.run(
                command,
                shell=True,
                capture_output=True,
                text=True,
                timeout=timeout,
            )
            return result.returncode, result.stdout
        except subprocess.TimeoutExpired:
            raise TimeoutError(f"Command timed out after {timeout}s")
```

def make_runner(behaviors: dict | None = None):
    """Create a mock command runner.

    behaviors: dict mapping step commands to callables or lists of (exit_code, output) per attempt.
    """
    attempt_counters: dict[str, int] = {}

    def runner(command: str, context: dict, timeout: float) -> tuple[int, str]:
        # Handle sleep commands -> simulate timeout
        if command.strip().startswith("sleep"):
            raise TimeoutError("timed out")

        # Handle exit codes
        if command.strip().startswith("exit"):
            code = int(command.strip().split()[-1])
            return (code, "")

        # Handle echo
        if command.strip().startswith("echo"):
            # Extract the string after echo, strip quotes
            text = command.strip()[5:].strip().strip('"').strip("'")
            return (0, text + "\n")

        # Handle custom behaviors
        if behaviors and command in behaviors:
            behavior = behaviors[command]
            attempt_counters.setdefault(command, 0)
            attempt_counters[command] += 1
            attempt = attempt_counters[command]
            if callable(behavior):
                return behavior(attempt, context, timeout)
            elif isinstance(behavior, list):
                idx = min(attempt - 1, len(behavior) - 1)
                return behavior[idx]

        return (0, "")

    return runner


def make_fail_then_succeed_runner(fail_counts: dict[str, int] | None = None):
    """Runner where fail_then_succeed(N) fails N times then succeeds.
    Also handles increment_counter().
    """
    attempt_counters: dict[str, int] = {}
    fail_counts = fail_counts or {}

    def runner(command: str, context: dict, timeout: float) -> tuple[int, str]:
        cmd = command.strip()

        if cmd.startswith("sleep"):
            raise TimeoutError("timed out")

        if cmd.startswith("exit"):
            code = int(cmd.split()[-1])
            return (code, "")

        if cmd.startswith("echo"):
            text = cmd[5:].strip().strip('"').strip("'")
            return (0, text + "\n")

        # fail_then_succeed(N) — fail N times, then succeed
        match = _match_fail_then_succeed(cmd)
        if match:
            n = int(match)
            attempt_counters.setdefault(cmd, 0)
            attempt_counters[cmd] += 1
            attempt = attempt_counters[cmd]
            if attempt <= n:
                return (1, f"attempt_{attempt}\n")
            return (0, f"attempt_{attempt}\n")

        # increment_counter()
        if "increment_counter" in cmd:
            attempt_counters.setdefault(cmd, 0)
            attempt_counters[cmd] += 1
            attempt = attempt_counters[cmd]
            if attempt == 1:
                return (1, f"attempt_{attempt}\n")
            return (0, f"attempt_{attempt}\n")

        return (0, "")

    return runner


def _match_fail_then_succeed(cmd: str) -> str | None:
    import re
    m = re.match(r"fail_then_succeed\((\d+)\)", cmd.strip())
    return m.group(1) if m else None

        assert result["results"]["step_b"].status == StepStatus.FAILED

    def test_diamond_one_branch_retried_one_timed_out(self):
        runner = make_fail_then_succeed_runner()
        executor = RecipeStepExecutor(command_runner=runner)
        recipe = [
            {"id": "step_a", "command": 'echo "root"'},

    def test_sub_recipe_propagated_feeds_conditional_parent_step(self):

    def test_chained_retries_step_b_uses_final_output(self):
        runner = make_fail_then_succeed_runner()

        recipe = [

        assert result["results"]["step_a"].status == StepStatus.COMPLETED

## Key Design Decisions

1. **Dependency before condition**: The spec requires that dependency evaluation happens BEFORE condition evaluation. A step blocked by a failed dep gets `failed`, not `skipped`, even if its condition would evaluate to false.

2. **Topological execution**: Steps are executed in waves — each wave picks up all steps whose dependencies have resolved. This naturally handles diamond graphs without explicit topological sort.

3. **Timeout = no retry**: On `TimeoutError`, the method returns immediately with `timed_out` status, short-circuiting the retry loop.

4. **Sub-recipe isolation**: Child context is a copy of parent. Writes only flow back to parent if `propagate_outputs` is true. Sub-recipe failures are non-transient and skip the retry mechanism.

5. **Template resolution**: `{{key}}` tokens referencing missing/skipped steps remain as literals — no silent empty substitution.

6. **Pluggable command runner**: Tests inject a mock runner to avoid real shell execution and control failure/success behavior precisely. The retry delay tests still use `time.sleep` for backoff verification.
