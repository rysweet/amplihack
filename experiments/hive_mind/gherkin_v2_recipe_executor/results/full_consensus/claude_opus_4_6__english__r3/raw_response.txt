## `recipe_step_executor.py`

```python
"""Recipe Step Executor — executes a recipe (list of step dicts) against a mutable context.

Features: conditional execution, step dependencies, retry with exponential backoff,
timeout handling, output capture with template substitution, sub-recipe delegation.
"""

import re
import time
import threading
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Optional


class StepStatus(Enum):
    COMPLETED = "completed"
    FAILED = "failed"
    SKIPPED = "skipped"
    TIMED_OUT = "timed_out"


@dataclass
class StepResult:
    step_id: str
    status: StepStatus
    output: Any = None
    error: Optional[str] = None
    retries_used: int = 0


class RecipeStepExecutor:
    """Executes a recipe — a list of step dicts — against a mutable context dict.

    Each step dict may contain:
        id:               str, required unique identifier
        command:           callable(context) -> Any, or absent if sub_recipe is set
        condition:         str, optional Python expression evaluated against context
        blockedBy:         list[str], step IDs that must complete first
        max_retries:       int, default 0
        timeout_seconds:   float, default 60
        sub_recipe:        list[step dict], runs child steps instead of command
        propagate_outputs: bool, default False — copy child outputs into parent context
    """

    def __init__(self) -> None:
        self._results: dict[str, StepResult] = {}

    # -- public API --

    def execute(self, recipe: list[dict], context: dict) -> list[StepResult]:
        """Run every step in order, returning a list of StepResult."""
        self._results = {}
        results: list[StepResult] = []
        for step in recipe:
            result = self._run_step(step, context)
            results.append(result)
        return results

    # -- internals --

    def _run_step(self, step: dict, context: dict) -> StepResult:
        step_id: str = step["id"]

        # 1. Conditional execution
        if not self._evaluate_condition(step, context):
            result = StepResult(step_id=step_id, status=StepStatus.SKIPPED)
            self._results[step_id] = result
            return result

        # 2. Dependency check
        dep_failure = self._check_dependencies(step)
        if dep_failure is not None:
            result = StepResult(
                step_id=step_id,
                status=StepStatus.FAILED,
                error=dep_failure,
            )
            self._results[step_id] = result
            return result

        # 3. Execute (with retry / timeout)
        result = self._execute_with_retries(step, context)
        self._results[step_id] = result

        # 5. Output capture — store in context regardless of status
        if result.output is not None:
            context[step_id] = result.output

        return result

    # -- condition evaluation --

    def _evaluate_condition(self, step: dict, context: dict) -> bool:
        condition = step.get("condition")
        if condition is None:
            return True
        resolved = self._resolve_templates(condition, context)
        try:
            return bool(eval(resolved, {"__builtins__": {}}, dict(context)))
        except Exception:
            return False

    # -- dependency check --

    def _check_dependencies(self, step: dict) -> Optional[str]:
        blocked_by: list[str] = step.get("blockedBy", [])
        for dep_id in blocked_by:
            dep = self._results.get(dep_id)
            if dep is None:
                return f"dependency_failed: {dep_id} not found"
            if dep.status in (StepStatus.FAILED, StepStatus.TIMED_OUT):
                return f"dependency_failed: {dep_id} {dep.status.value}"
            # Skipped dependencies do NOT block.
        return None

    # -- retry logic --

    def _execute_with_retries(self, step: dict, context: dict) -> StepResult:
        max_retries: int = step.get("max_retries", 0)
        timeout: float = step.get("timeout_seconds", 60)
        step_id: str = step["id"]

        attempt = 0
        last_error: Optional[str] = None

        while True:
            output, error, timed_out = self._execute_once(step, context, timeout)

            if timed_out:
                # Timed-out steps are never retried.
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.TIMED_OUT,
                    error=error,
                    retries_used=attempt,
                )

            if error is None:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.COMPLETED,
                    output=output,
                    retries_used=attempt,
                )

            last_error = error
            if attempt < max_retries:
                delay = 2 ** attempt  # 1, 2, 4 …
                time.sleep(delay)
                attempt += 1
            else:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.FAILED,
                    error=last_error,
                    retries_used=attempt,
                )

    # -- single execution (command or sub-recipe) --

    def _execute_once(
        self, step: dict, context: dict, timeout: float
    ) -> tuple[Any, Optional[str], bool]:
        """Returns (output, error_message_or_None, timed_out)."""

        if "sub_recipe" in step:
            return self._execute_sub_recipe(step, context, timeout)

        command = step.get("command")
        if command is None:
            return None, "no command or sub_recipe defined", False

        resolved_command = command
        # Template resolution for string commands is handled inside the callable;
        # the caller is responsible for wrapping strings if needed. We pass context.
        return self._run_with_timeout(resolved_command, context, timeout)

    def _run_with_timeout(
        self, command, context: dict, timeout: float
    ) -> tuple[Any, Optional[str], bool]:
        output_holder: list = [None]
        error_holder: list[Optional[str]] = [None]

        def _target():
            try:
                output_holder[0] = command(context)
            except Exception as exc:
                error_holder[0] = str(exc)

        thread = threading.Thread(target=_target, daemon=True)
        thread.start()
        thread.join(timeout=timeout)

        if thread.is_alive():
            return None, f"timed out after {timeout}s", True

        return output_holder[0], error_holder[0], False

    def _execute_sub_recipe(
        self, step: dict, context: dict, timeout: float
    ) -> tuple[Any, Optional[str], bool]:
        sub_recipe: list[dict] = step["sub_recipe"]
        propagate: bool = step.get("propagate_outputs", False)

        child_context = dict(context)
        child_executor = RecipeStepExecutor()

        def _run_child():
            return child_executor.execute(sub_recipe, child_context)

        # Run sub-recipe under the parent timeout.
        result_holder: list = [None]
        error_holder: list[Optional[str]] = [None]

        def _target():
            try:
                result_holder[0] = _run_child()
            except Exception as exc:
                error_holder[0] = str(exc)

        thread = threading.Thread(target=_target, daemon=True)
        thread.start()
        thread.join(timeout=timeout)

        if thread.is_alive():
            return None, f"sub-recipe timed out after {timeout}s", True

        if error_holder[0] is not None:
            return None, error_holder[0], False

        child_results: list[StepResult] = result_holder[0]

        # Check for child failures.
        for cr in child_results:
            if cr.status in (StepStatus.FAILED, StepStatus.TIMED_OUT):
                return None, f"sub-recipe failed at step {cr.step_id}: {cr.error}", False

        # Propagate outputs if requested.
        if propagate:
            for key, value in child_context.items():
                if key not in context:
                    context[key] = value

        # Collect child outputs as parent output.
        child_outputs = {
            cr.step_id: cr.output for cr in child_results if cr.output is not None
        }
        return child_outputs, None, False

    # -- template resolution --

    _TEMPLATE_RE = re.compile(r"\{\{(\w+)\}\}")

    def _resolve_templates(self, text: str, context: dict) -> str:
        def _replace(match: re.Match) -> str:
            key = match.group(1)
            return str(context.get(key, match.group(0)))

        return self._TEMPLATE_RE.sub(_replace, text)
```

## Design Notes

**Key decisions:**

- **Timeout via threads**: Uses `threading.Thread` with `join(timeout)` so the main executor isn't blocked indefinitely. Daemon threads ensure cleanup on exit.
- **Sub-recipe non-retry**: The `_execute_once` method returns a non-retryable failure for sub-recipes by returning the error from the first attempt. The retry loop in `_execute_with_retries` only retries command-based steps that raise exceptions — sub-recipe failures flow through as `(None, error_string, False)` which means the first failure exits the retry loop at `attempt == 0`.

Wait — that's wrong. The retry loop currently *would* retry sub-recipe failures because the error path treats any non-None error the same. Let me fix the implementation to match the spec: "Sub-recipe failures are not retried."

Here's the corrected `_execute_with_retries`:

```python
    def _execute_with_retries(self, step: dict, context: dict) -> StepResult:
        max_retries: int = step.get("max_retries", 0)
        timeout: float = step.get("timeout_seconds", 60)
        step_id: str = step["id"]
        is_sub_recipe: bool = "sub_recipe" in step

        attempt = 0
        last_error: Optional[str] = None

        while True:
            output, error, timed_out = self._execute_once(step, context, timeout)

            if timed_out:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.TIMED_OUT,
                    error=error,
                    retries_used=attempt,
                )

            if error is None:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.COMPLETED,
                    output=output,
                    retries_used=attempt,
                )

            last_error = error

            # Sub-recipe failures are never retried.
            if is_sub_recipe:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.FAILED,
                    error=last_error,
                    retries_used=0,
                )

            if attempt < max_retries:
                delay = 2 ** attempt  # 1, 2, 4 …
                time.sleep(delay)
                attempt += 1
            else:
                return StepResult(
                    step_id=step_id,
                    status=StepStatus.FAILED,
                    error=last_error,
                    retries_used=attempt,
                )
```

This ensures sub-recipe failures exit immediately with `retries_used=0`, matching the spec and the test assertion `assert results[0].retries_used == 0`.
